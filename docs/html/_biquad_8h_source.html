22 class Biquad : public InsertEffect {
23 public:
24 virtual void updateCoeffs() = 0;
25
26 Biquad(const float fs, const int channels) : InsertEffect(fs, channels) {
27 f0 = 1000.f;
28 Q = 0.7071f;
29 A = 1.0f;
30 w0 = 0; a = 0; a0 = 0; a1 = 0; a2 = 0;
31 b0 = 1; b1 = 0; b2 = 0;
32 for (auto&[xnm1, xnm2, ynm1, ynm2] : state) {
33 xnm1 = 0;
34 xnm2 = 0;
35 ynm1 = 0;
36 ynm2 = 0;
37 }
38 }
39
40 void reset() override {
41 for (auto&[xnm1, xnm2, ynm1, ynm2] : state) {
42 xnm1 = 0;
43 xnm2 = 0;
44 ynm1 = 0;
45 ynm2 = 0;
46 }
47 updateCoeffs();
48 }
49
50 void setFreq(const float _frequency) {
51 f0 = std::clamp(_frequency, 0.f, fs/2 - 100);
52 updateCoeffs();
53 }
54 void setQ(const float _q) {
55 Q = _q;
56 updateCoeffs();
57 }
58 void setA(const float _a) {
59 A = _a;
60 updateCoeffs();
61 }
62 void setAll(const float _frequency, const float _q, const float _a) {
63 f0 = _frequency;
64 Q = _q;
65 A = _a;
66 updateCoeffs();
67 }
68
69 float getFreq() { return f0; }
70 float getQ() { return Q; }
71 float getA() { return A; }
72
73 float processSample(const float xn, const int channel) override {
74 if (channel >= MAX_CHANNELS) { return 0.f; }
75
76 const float yn = (a0/b0)*xn + (a1/b0)*state.at(channel).xnm1 + (a2/b0)*state.at(channel).xnm2
77 - (b1/b0)*state.at(channel).ynm1 - (b2/b0)*state.at(channel).ynm2;
78
79 state.at(channel).xnm2 = state.at(channel).xnm1;
80 state.at(channel).xnm1 = xn;
81 state.at(channel).ynm2 = state.at(channel).ynm1;
82 state.at(channel).ynm1 = yn;
83 return yn;
84 }
85
86
87 protected:
88 struct State {
89 float xnm1{}, xnm2{}, ynm1{}, ynm2{};
90 };
91
92 float f0{}, Q{}, A{};
93 float w0{}, a{}, a0{}, a1{}, a2{}, b0{}, b1{}, b2{};
94 std::array<State, MAX_CHANNELS> state;
95 };

99 class RBJ : public Biquad {
100 public:
101
102 RBJ(const float fs, const int channels) : Biquad(fs, channels) {
103 Biquad::reset();
104 }
105
106 void setFilterType(BiquadType _filterType) {
107 filterType = _filterType;
108 updateCoeffs();
109 }
110
111 void updateCoeffs() override {
112 w0 = (2*M_PI*f0)/fs;
113 a = sin(w0)/(2*Q);
114
115 float cosw0 = cos(w0);
116
117 switch (filterType) {
118 case BiquadType::LOWPASS:
119 a0 = (1-cosw0)/2;
120 a1 = 1-cosw0;
121 a2 = a0;
122 b0 = 1+a;
123 b1 = -2*cosw0;
124 b2 = 1-a;
125 break;
126 case BiquadType::HIGHPASS:
127 a0 = (1+cosw0)/2;
128 a1 = -1 * (1+cosw0);
129 a2 = a0;
130 b0 = 1+a;
131 b1 = -2*cosw0;
132 b2 = 1-a;
133 break;
134 case BiquadType::BANDPASS:
135 a0 = A * a;
136 a1 = 0;
137 a2 = -A * a;
138 b0 = 1 + a;
139 b1 = -2 * cosw0;
140 b2 = 1 - a;
141 break;
142 case BiquadType::NOTCH:
143 a0 = 1;
144 a1 = -2 * cosw0;
145 a2 = 1;
146 b0 = 1 + a;
147 b1 = a1;
148 b2 = 1 - a;
149 break;
150 case BiquadType::PEAKING:
151 a0 = 1 + A * a;
152 a1 = -2 * cosw0;
153 a2 = 1 - A * a;
154 b0 = 1 + (a/A);
155 b1 = a1;
156 b2 = 1 - (a/A);
157 break;
158 case BiquadType::HIGH_SHELF:
159 a0 = A * ((A+1)+(A-1) * cosw0 + 2*sqrt(A)*a);
160 a1 = -2*A*((A-1)+(A+1)*cosw0);
161 a2 = A * ((A+1)+(A-1) * cosw0 - 2*sqrt(A)*a);
162 b0 = (A+1) - (A-1) * cosw0 + 2*sqrt(A)*a;
163 b1 = 2 * ((A-1)-(A+1)*cosw0);
164 b2 = (A+1) - (A-1) * cosw0 - 2*sqrt(A)*a;
165 break;
166 case BiquadType::LOW_SHELF:
167 a0 = A * ((A+1)-(A-1) * cosw0 + 2*sqrt(A)*a);
168 a1 = 2*A*((A-1)-(A+1)*cosw0);
169 a2 = A * ((A+1)-(A-1) * cosw0 - 2*sqrt(A)*a);
170 b0 = (A+1) + (A-1) * cosw0 + 2*sqrt(A)*a;
171 b1 = -2 * ((A-1)+(A+1)*cosw0);
172 b2 = (A+1) + (A-1) * cosw0 - 2*sqrt(A)*a;
173 break;
174 case BiquadType::ALL_PASS:
175 a0 = 1-a;
176 a1 = -2*cosw0;
177 a2 = 1+a;
178 b0 = a2;
179 b1 = a1;
180 b2 = a0;
181 break;
182 default:
183 a0 = (1-cosw0)/2;
184 a1 = 1-cosw0;
185 a2 = a0;
186 b0 = 1+a;
187 b1 = -2*cosw0;
188 b2 = 1-a;
189 break;
190 }
191 }
192 protected:
193 BiquadType filterType = BiquadType::LOWPASS;
194 };
