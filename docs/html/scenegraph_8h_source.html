151 inline Transform SceneGraph::worldTransform(const EntityID id, const Registry& registry) {
152 // Return cached result if clean.
153 if (_dirty.count(id) == 0 && _worldCache.count(id)) {
154 const auto it = _worldCache.find(id);
155 if (it != _worldCache.end()) return it->second;
156 }
157
158 const Transform local = registry.hasComponent<Transform>(id)
159 ? registry.getComponent<Transform>(id)
160 : Transform{};
161
162 Transform world = local;
163 if (hasParent(id))
164 world = worldTransform(_parents.at(id), registry) * local;
165
166 _worldCache[id] = world;
167 _dirty.erase(id);
168 return world;
169 }

175 inline void SceneGraph::remove(const EntityID id) {
176 // Detach children — they become root nodes.
177 if (_children.find(id) != _children.end()) {
178 for (const EntityID child : _children.at(id))
179 _parents.erase(child);
180 _children.erase(id);
181 }
182 detach(id);
183 _worldCache.erase(id);
184 _dirty.erase(id);
185 }

187 inline void SceneGraph::_invalidateRecursive(const EntityID id) {
188 _dirty.insert(id);
189 _worldCache.erase(id);
190 if (_children.find(id) == _children.end()) return;
191 for (const EntityID child : _children.at(id))
192 _invalidateRecursive(child);
193 }
