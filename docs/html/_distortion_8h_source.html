10namespace Quartz {
11 class Distortion : public InsertEffect {
12 public:
13 Distortion(const float _fs, const int numChannels) : InsertEffect(_fs, numChannels) {};
14
15 virtual float distortion (float xn) = 0;
16
17 float processSample(float xn, int channel) override {
18 return distortion(applyDriveBias(xn));
19 }
20
21 void setDrive (float _driveMag) { driveMag = _driveMag; }
22 void setBias (float _bias) { bias = _bias; }
23
24 float getDriveMag () const { return driveMag; }
25 float getBias () const { return bias; }
26 protected:
27 float applyDriveBias(const float xn) const {
28 return xn * driveMag + bias;
29 }
30
31 float driveMag = 1.f;
32 float bias = 0.f;
33 };

83 class SoftClipper : public Distortion {
84 public:
85 SoftClipper(const float fs, const int numChannels) : Distortion(fs, numChannels) {
86 setThreshold_linear(1.0);
87 setKnee_dB(3.0);
88 }
89
90 void setThreshold_linear(double threshold) {
91 threshold_linear = threshold;
92 }
93
94 void setKnee_dB(double _knee_dB) {
95 knee_dB = _knee_dB;
96 knee_linear = std::min((2*threshold_linear), (1.0 / std::pow(10.0f, _knee_dB / 20.0f)) - 1.0);
97 //knee_linear = 2.0 * threshold_linear * std::pow(10.0, knee_dB / 20.0) - 2.0 * threshold_linear;
98 halfKnee = 0.5 * knee_linear;
99 }
100
101 double getThreshold_linear() const { return threshold_linear; }
102 double getKnee_dB() const { return knee_dB; }
103
104 float distortion(float xn) override {
105 const float abs = std::fabs(xn);
106 const int sign = xn >= 0 ? 1 : -1;
107
108 if (abs <= threshold_linear - halfKnee) {
109 return xn;
110 } else if (abs <= threshold_linear + halfKnee) {
111 const float excess = abs - (threshold_linear - halfKnee);
112 return sign * (abs - (excess * excess) / (2 * knee_linear));
113 } else {
114 return sign * threshold_linear;
115 }
116 }
117 private:
118 double threshold_linear{};
119 double knee_linear{};
120 double knee_dB{};
121 double halfKnee{};
122 };
