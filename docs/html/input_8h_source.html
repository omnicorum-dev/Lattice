453inline char keycodeToChar(const int keycode, const int mods)
454{
455 const bool shift = (mods & MED_MOD_SHIFT) != 0;
456 const bool caps = (mods & MED_MOD_CAPS_LOCK) != 0;
457
458 // Letters: Shift XOR Caps Lock = uppercase
459 if (keycode >= MED_KEY_A && keycode <= MED_KEY_Z)
460 {
461 bool upper = shift ^ caps;
462 return static_cast<char>((upper ? 'A' : 'a') + (keycode - MED_KEY_A));
463 }
464
465 // Number row and punctuation
466 switch (keycode)
467 {
468 case MED_KEY_0: return shift ? ')' : '0';
469 case MED_KEY_1: return shift ? '!' : '1';
470 case MED_KEY_2: return shift ? '@' : '2';
471 case MED_KEY_3: return shift ? '#' : '3';
472 case MED_KEY_4: return shift ? '$' : '4';
473 case MED_KEY_5: return shift ? '%' : '5';
474 case MED_KEY_6: return shift ? '^' : '6';
475 case MED_KEY_7: return shift ? '&' : '7';
476 case MED_KEY_8: return shift ? '*' : '8';
477 case MED_KEY_9: return shift ? '(' : '9';
478
479 case MED_KEY_SPACE: return ' ';
480 case MED_KEY_APOSTROPHE: return shift ? '"' : '\'';
481 case MED_KEY_COMMA: return shift ? '<' : ',';
482 case MED_KEY_MINUS: return shift ? '_' : '-';
483 case MED_KEY_PERIOD: return shift ? '>' : '.';
484 case MED_KEY_SLASH: return shift ? '?' : '/';
485 case MED_KEY_SEMICOLON: return shift ? ':' : ';';
486 case MED_KEY_EQUAL: return shift ? '+' : '=';
487 case MED_KEY_LEFT_BRACKET: return shift ? '{' : '[';
488 case MED_KEY_BACKSLASH: return shift ? '|' : '\\';
489 case MED_KEY_RIGHT_BRACKET: return shift ? '}' : ']';
490 case MED_KEY_GRAVE_ACCENT: return shift ? '~' : '`';
491
492 // Keypad (Num Lock assumed on)
493 case MED_KEY_KP_0: return '0';
494 case MED_KEY_KP_1: return '1';
495 case MED_KEY_KP_2: return '2';
496 case MED_KEY_KP_3: return '3';
497 case MED_KEY_KP_4: return '4';
498 case MED_KEY_KP_5: return '5';
499 case MED_KEY_KP_6: return '6';
500 case MED_KEY_KP_7: return '7';
501 case MED_KEY_KP_8: return '8';
502 case MED_KEY_KP_9: return '9';
503 case MED_KEY_KP_DECIMAL: return '.';
504 case MED_KEY_KP_DIVIDE: return '/';
505 case MED_KEY_KP_MULTIPLY: return '*';
506 case MED_KEY_KP_SUBTRACT: return '-';
507 case MED_KEY_KP_ADD: return '+';
508 case MED_KEY_KP_EQUAL: return '=';
509
510 default: return '\0';
511 }
