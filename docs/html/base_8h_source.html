194 static f32 catmullRomSpline(const f32 p0, const f32 p1, const f32 p2, const f32 p3, const f32 t) {
195 const f32 t2 = t * t;
196 const f32 t3 = t2 * t;
197 return 0.5f * (
198 2.0f * p1 +
199 (-p0 + p2) * t +
200 (2.0f * p0 - 5.0f * p1 + 4.0f * p2 - p3) * t2 +
201 (-p0 + 3.0f * p1 - 3.0f * p2 + p3) * t3
202 );
203 }

256 inline std::string fStringPrint(std::stringstream& ss, const std::string& fmt, size_t pos) {
257 if (fmt.find("{}", pos) != std::string::npos) {
258 throw std::runtime_error("Too few arguments for format string");
259 }

261 return ss.str();

276 template<typename T, typename... Args>
277 std::string fStringPrint(std::stringstream& ss, const std::string& fmt, size_t pos, T value, Args... args) {
278 const size_t brace = fmt.find("{}", pos);
279 if (brace == std::string::npos) {
280 throw std::runtime_error("Too many arguments for format string");
281 }
282 ss << fmt.substr(pos, brace - pos);
283 ss << value;
284 return fStringPrint(ss, fmt, brace + 2, args...);
285 }
